  description="A controlled and uncontrolled checkbox with support for sizes and disabled state."
/>

## Default

An uncontrolled checkbox with no initial value.

<Preview>
  <Checkbox />
</Preview>

<CodeBlock className="mt-6"
  language="tsx"
  code={`<Checkbox />`}
/>

## Checked

Use `defaultChecked` to set the initial checked state in uncontrolled mode.

<Preview>
  <Checkbox defaultChecked />
</Preview>

<CodeBlock className="mt-6"
  language="tsx"
  code={`<Checkbox defaultChecked />`}
/>

## Sizes

Use the `size` prop to control the checkbox dimensions. The default size is `md`.

<Preview>
  <div className="flex gap-4 items-center">
    <Checkbox size="sm" />
    <Checkbox size="md" />
  </div>
</Preview>

<CodeBlock className="mt-6"
  language="tsx"
  code={`<Checkbox size="sm" />
<Checkbox size="md" />`}
/>

## Disabled

Pass the `disabled` prop to prevent interaction.

<Preview>
  <div className="flex gap-4 items-center">
    <Checkbox disabled />
    <Checkbox defaultChecked disabled />
  </div>
</Preview>

<CodeBlock className="mt-6"
  language="tsx"
  code={`<Checkbox disabled />
<Checkbox defaultChecked disabled />`}
/>
